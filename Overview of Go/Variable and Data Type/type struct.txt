type User struct{
	firstName 	string
	lastName 	string
}

penggunaan:
user := User {
	firstName : "Farid",
	lastName : "Nubaili",
}

var user User

user.firstName = "farid"

kita juga dapat menyertakan sebuah fungsi pada pendefinisian struct, menggunakan receiver
func (m *myStruct) func_name() string{
	return m.firstName
}

// Struct dengan data
person := struct {
	Name      string `json:"name"`
	Age       int    `json:"age"`
	IsMarried bool   `json:"isMarried"`
}{
	Name:      "Alice",
	Age:       25,
	IsMarried: true,
}
